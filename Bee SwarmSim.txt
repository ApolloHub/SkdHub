game:GetService("RunService").Stepped:Connect(
    function()
        spawn(
            function()
                if autofarm == true then
                for i, v in next, game.Workspace[lcalplr.Name]:GetChildren() do
                    if v:IsA("Tool") then
                        for g, c in next, v:GetChildren() do
                            if c.Name == "ClickEvent" and c:IsA("RemoteEvent") then
                                game.workspace[lcalplr.Name][v.Name].ClickEvent:FireServer()
                            end
                        end
                    end
                end
            end
            end
        )
    end
)

while wait() do
    if autofarm == true then
    for i, v in next, game.Workspace[lcalplr.Name]:GetChildren() do
        if autosell == true then
        if v:IsA("Accessory") then
            for g, c in next, v:GetChildren() do
                if c.Name == "Display" and c:IsA("Part") then
                    local splited = c.Gui.ProgressLabel.Text:split("/")
                    if splited[1] == splited[2] then
                        if ismakinghoney == false then
                        tptobase()
                        end
                        ismakinghoney = true
                    end
                end
            end
        end
    end
    end

    if ismakinghoney == true then
        for i, v in next, game.Workspace[lcalplr.Name]:GetChildren() do
            if v:IsA("Accessory") then
                for g, c in next, v:GetChildren() do
                    if c.Name == "Display" and c:IsA("Part") then
                        local splited = c.Gui.ProgressLabel.Text:split("/")
                        if splited[1] == "0" then
                            wait(5)
                            ismakinghoney = false
                            tptofield()
                        end
                    end
                end
            end
        end
    end
	end
	if getgenv().collecttoken == true then
        for i,v in pairs(game.Workspace.Collectibles:GetChildren()) do
            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.CFrame
        end
    end
    if getgenv().mobautofarm == true then
        game:GetService('RunService').Stepped:connect(function()
            wait()
            if getgenv().mobautofarm == true then
                for i,v in pairs(game:GetService("Workspace").Monsters:GetChildren()) do
                    for i,v in pairs(v:GetChildren()) do
                        if v:IsA("Part") then
                            game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = v.CFrame * CFrame.new(0, 10, 0)
                        end
                    end
                end
            else
                print("no")
                return nil
            end
        end)
    end
    if getgenv().autoclicker == true then
        mouse1click()  
    end
    if getgenv().snwoflakefarm == true then
        for i,v in pairs(game:GetService("Workspace").Collectibles:GetChildren()) do
            for i,v in pairs(v:GetChildren()) do
                if v.Name == "FrontDecal" then
                    if v.Texture == "rbxassetid://6087969886" then
                        game:GetService("Players")["LocalPlayer"].Character.HumanoidRootPart.CFrame = v.Parent.CFrame
                    end
                end
            end
        end
    end 
end
